QUANTS = ["Q2_K","Q3_K","Q4_K_M",
         "Q5_K_M","Q6_K","Q8_0"]

class Q8AugmentedPolicy:
  """UCB1 with Q8 priority boost"""
  def __init__(self):
    self.counts = {q:0 for q in QUANTS}
    self.rewards = {q:0.0 for q in QUANTS}
    self.q8_boost = 0.15 # Q8 accuracy bonus

  def select(self, n):
    scores = {}for q in QUANTS:
      exploit = self.rewards[q]
      explore = sqrt(2*log(n)/self.counts[q])
      # Q8 priority augmentation
      boost = self.q8_boost if q=="Q8_0" else 0
      scores[q] = exploit + explore + boost
    return max(scores, key=scores.get)
