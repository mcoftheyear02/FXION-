#!/usr/bin/env python3
# qint2_quant.py
# Advanced QINT2 quantizer for FXION-ONYX

from typing import List, Tuple
import math

Q_TO_BITS = {
    -3: "00",
    -1: "01",
     1: "10",
     3: "11",
}

LEVELS = [-3, -1, 1, 3]


def choose_scale(block: List[float]) -> float:
    max_abs = max(abs(x) for x in block) if block else 1.0
    return 1.0 if max_abs == 0 else max_abs / 3.0


def quantize_value(x: float, scale: float) -> Tuple[int, str]:
    q_star = round(x / scale)
    q = min(LEVELS, key=lambda v: abs(v - q_star))
    return q, Q_TO_BITS[q]


def quantize_block(block: List[float]) -> Tuple[float, List[int], List[str]]:
    if not block:
        return 1.0, [], []

    scale = choose_scale(block)
    qs = []
    bits_list = []

    for x in block:
        q, bits = quantize_value(x, scale)
        qs.append(q)
        bits_list.append(bits)

    return scale, qs, bits_list


if __name__ == "__main__":
    values = [0.1, -0.5, 0.9, -1.2]
    scale, qs, bits = quantize_block(values)

    print("values:", values)
    print("scale :", scale)
    print("q     :", qs)
    print("bits  :", bits)
